    ID INT,
    nome VARCHAR(40),

CREATE TABLE tb_filmes (
    ID INT,
    titulo VARCHAR(40),
    duracao VARCHAR(20), 
    idioma_original VARCHAR(20),
    preco DECIMAL(8, 2),
);

CREATE TABLE tb_elenco (
    ID INT,
    nome VARCHAR(40),
    dt_nasc DATE,
    nacionalidade VARCHAR(20),
    filme VARCHAR(40),
);
